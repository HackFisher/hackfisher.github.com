// DAOTemplate.java:
public static execute(Action action) {
DAOSession session = null;
try {
     session = createSession();
     session.beginTransaction();
     action.execute(session);
} catch (Exception e) {
     rollbackSession(session);
     throw e;
} finally {
     closeSession(session);
}
}
